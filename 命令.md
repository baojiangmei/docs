### git 命令
 1. 源码安装
    $ tar -zxf git-2.8.0.tar.gz
    
    $ cd git-2.8.0
    
    $ make configure
    
    $ ./configure --prefix=/usr
    
    $ make all doc info
    
    $ sudo make install install-doc install-html install-info
 3. 下载代码 git clone git://git.kernel.org/pub/scm/git/git.git -b version
